-- Criar a tabela
DROP TABLE IF EXISTS casos_feminicidio;

CREATE TABLE casos_feminicidio (
    ano INTEGER NOT NULL,
    estado TEXT NOT NULL,
    quantidade INTEGER NOT NULL,
    fonte TEXT
);

-- Conferir os dados
SELECT ano, estado, quantidade
FROM casos_feminicidio
WHERE ano = 2025;

-- Soma dos estados (sem Brasil e TOTAL GERAL)
SELECT SUM(quantidade) AS soma_estados
FROM casos_feminicidio
WHERE ano = 2025
  AND estado NOT IN ('Brasil','TOTAL GERAL');

-- Percentual por estado
SELECT estado,
       quantidade,
       ROUND((quantidade * 100.0) / 
             (SELECT quantidade FROM casos_feminicidio WHERE ano = 2025 AND estado = 'TOTAL GERAL'),2) AS percentual
FROM casos_feminicidio
WHERE ano = 2025
  AND estado NOT IN ('Brasil','TOTAL GERAL')
ORDER BY percentual DESC;
